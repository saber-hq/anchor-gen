6//! transaction than the legacy format.

8//! [`legacy`]: crate::legacy

96            temp_table
97        };

113fn position(keys: &[Pubkey], key: &Pubkey) -> u8 {
114    keys.iter().position(|k| k == key).unwrap() as u8

117fn compile_instruction(ix: &Instruction, keys: &[Pubkey]) -> CompiledInstruction {
118    let accounts: Vec<_> = ix
119        .accounts
120        .iter()
121        .map(|account_meta| position(keys, &account_meta.pubkey))
122        .collect();
123
124    CompiledInstruction {
125        program_id_index: position(keys, &ix.program_id),
126        data: ix.data.clone(),
127        accounts,
128    }
129}
130
131fn compile_instructions(ixs: &[Instruction], keys: &[Pubkey]) -> Vec<CompiledInstruction> {
132    ixs.iter().map(|ix| compile_instruction(ix, keys)).collect()
133}

135/// A Solana transaction message (legacy).
136///
137/// See the crate documentation for further description.
138///
139/// Some constructors accept an optional `payer`, the account responsible for
140/// paying the cost of executing a transaction. In most cases, callers should
141/// specify the payer explicitly in these constructors. In some cases though,
142/// the caller is not _required_ to specify the payer, but is still allowed to:
143/// in the `Message` structure, the first account is always the fee-payer, so if
144/// the caller has knowledge that the first account of the constructed
145/// transaction's `Message` is both a signer and the expected fee-payer, then
146/// redundantly specifying the fee-payer is not strictly required.
147// NOTE: Serialization-related changes must be paired with the custom serialization
148// for versioned messages in the `RemainingLegacyMessage` struct.
149#[cfg(not(target_arch = "wasm32"))]
150#[cfg_attr(
151    feature = "frozen-abi",
152    frozen_abi(digest = "2THeaWnXSGDTsiadKytJTcbjrk4KjfMww9arRLZcwGnw"),
153    derive(AbiExample)
154)]
155#[cfg_attr(
156    feature = "serde",
157    derive(Deserialize, Serialize),
158    serde(rename_all = "camelCase")
159)]
160#[derive(Default, Debug, PartialEq, Eq, Clone)]
161pub struct Message {
162    /// The message header, identifying signed and read-only `account_keys`.
163    // NOTE: Serialization-related changes must be paired with the direct read at sigverify.
164    pub header: MessageHeader,
165
166    /// All the account keys used by this transaction.
167    #[cfg_attr(feature = "serde", serde(with = "solana_short_vec"))]
168    pub account_keys: Vec<Pubkey>,
169
170    /// The id of a recent ledger entry.
171    pub recent_blockhash: Hash,
172
173    /// Programs that will be executed in sequence and committed in one atomic transaction if all
174    /// succeed.
175    #[cfg_attr(feature = "serde", serde(with = "solana_short_vec"))]
176    pub instructions: Vec<CompiledInstruction>,
177}

211impl Sanitize for Message {
212    fn sanitize(&self) -> std::result::Result<(), SanitizeError> {
213        // signing area and read-only non-signing area should not overlap
214        if self.header.num_required_signatures as usize
215            + self.header.num_readonly_unsigned_accounts as usize
216            > self.account_keys.len()
217        {
218            return Err(SanitizeError::IndexOutOfBounds);
219        }
220
221        // there should be at least 1 RW fee-payer account.
222        if self.header.num_readonly_signed_accounts >= self.header.num_required_signatures {
223            return Err(SanitizeError::IndexOutOfBounds);
224        }
225
226        for ci in &self.instructions {
227            if ci.program_id_index as usize >= self.account_keys.len() {
228                return Err(SanitizeError::IndexOutOfBounds);
229            }
230            // A program cannot be a payer.
231            if ci.program_id_index == 0 {
232                return Err(SanitizeError::IndexOutOfBounds);
233            }
234            for ai in &ci.accounts {
235                if *ai as usize >= self.account_keys.len() {
236                    return Err(SanitizeError::IndexOutOfBounds);
237                }
238            }
239        }
240        self.account_keys.sanitize()?;
241        self.recent_blockhash.sanitize()?;
242        self.instructions.sanitize()?;
243        Ok(())
244    }
245}

346    /// // the client.
347    /// #[derive(BorshSerialize, BorshDeserialize)]
348    /// # #[borsh(crate = "borsh")]

364    ///         program_id,

369    ///     let blockhash = client.get_latest_blockhash()?;
370    ///
371    ///     let message = Message::new_with_blockhash(
372    ///         &[instruction],
373    ///         Some(&payer.pubkey()),
374    ///         &blockhash,
375    ///     );
376    ///
377    ///     let mut tx = Transaction::new_unsigned(message);
378    ///     tx.sign(&[payer], tx.message.recent_blockhash);
379    ///     client.send_and_confirm_transaction(&tx)?;
380    ///
381    ///     Ok(())
382    /// }
383    /// #
384    /// # let client = RpcClient::new(String::new());
385    /// # let program_id = Pubkey::new_unique();
386    /// # let payer = Keypair::new();
387    /// # send_initialize_tx(&client, program_id, &payer)?;
388    /// #
389    /// # Ok::<(), anyhow::Error>(())
390    /// ```
391    pub fn new_with_blockhash(
392        instructions: &[Instruction],
393        payer: Option<&Pubkey>,
394        blockhash: &Hash,
395    ) -> Self {
396        let compiled_keys = CompiledKeys::compile(instructions, payer.cloned());
397        let (header, account_keys) = compiled_keys
398            .try_into_message_components()
399            .expect("overflow when compiling message keys");
400        let instructions = compile_instructions(instructions, &account_keys);
401        Self::new_with_compiled_instructions(
402            header.num_required_signatures,
403            header.num_readonly_signed_accounts,
404            header.num_readonly_unsigned_accounts,
405            account_keys,
406            *blockhash,
407            instructions,
408        )
409    }

412    ///
413    /// [nonced transaction]: https://docs.solanalabs.com/implemented-proposals/durable-tx-nonces

416    /// transaction nonce_, allowing for extended time to pass between the

430    /// use anyhow::Result;

443    /// // A custom program instruction. This would typically be defined in

454    /// // Create a nonced transaction for later signing and submission,
455    /// // returning it and the nonce account's pubkey.
456    /// fn create_offline_initialize_tx(
457    ///     client: &RpcClient,
458    ///     program_id: Pubkey,
459    ///     payer: &Keypair
460    /// ) -> Result<(Transaction, Pubkey)> {
461    ///
462    ///     let bank_instruction = BankInstruction::Initialize;
463    ///     let bank_instruction = Instruction::new_with_borsh(
464    ///         program_id,
465    ///         &bank_instruction,
466    ///         vec![],
467    ///     );
468    ///
469    ///     // This will create a nonce account and assign authority to the
470    ///     // payer so they can sign to advance the nonce and withdraw its rent.
471    ///     let nonce_account = make_nonce_account(client, payer)?;
472    ///
473    ///     let mut message = Message::new_with_nonce(
474    ///         vec![bank_instruction],
475    ///         Some(&payer.pubkey()),
476    ///         &nonce_account,
477    ///         &payer.pubkey()
478    ///     );
479    ///
480    ///     // This transaction will need to be signed later, using the blockhash
481    ///     // stored in the nonce account.
482    ///     let tx = Transaction::new_unsigned(message);
483    ///
484    ///     Ok((tx, nonce_account))
485    /// }
486    ///
487    /// fn make_nonce_account(client: &RpcClient, payer: &Keypair)
488    ///     -> Result<Pubkey>
489    /// {
490    ///     let nonce_account_address = Keypair::new();
491    ///     let nonce_account_size = solana_nonce::state::State::size();
492    ///     let nonce_rent = client.get_minimum_balance_for_rent_exemption(nonce_account_size)?;
493    ///
494    ///     // Assigning the nonce authority to the payer so they can sign for the withdrawal,
495    ///     // and we can throw away the nonce address secret key.
496    ///     let create_nonce_instr = create_nonce_account(
497    ///         &payer.pubkey(),
498    ///         &nonce_account_address.pubkey(),
499    ///         &payer.pubkey(),
500    ///         nonce_rent,
501    ///     );
502    ///
503    ///     let mut nonce_tx = Transaction::new_with_payer(&create_nonce_instr, Some(&payer.pubkey()));
504    ///     let blockhash = client.get_latest_blockhash()?;
505    ///     nonce_tx.sign(&[&payer, &nonce_account_address], blockhash);
506    ///     client.send_and_confirm_transaction(&nonce_tx)?;
507    ///
508    ///     Ok(nonce_account_address.pubkey())
509    /// }
510    /// #
511    /// # let client = RpcClient::new(String::new());
512    /// # let program_id = Pubkey::new_unique();
513    /// # let payer = Keypair::new();
514    /// # create_offline_initialize_tx(&client, program_id, &payer)?;
515    /// # Ok::<(), anyhow::Error>(())
516    /// ```
517    #[cfg(feature = "bincode")]
518    pub fn new_with_nonce(
519        mut instructions: Vec<Instruction>,
520        payer: Option<&Pubkey>,
521        nonce_account_pubkey: &Pubkey,
522        nonce_authority_pubkey: &Pubkey,
523    ) -> Self {
524        let nonce_ix = solana_system_interface::instruction::advance_nonce_account(
525            nonce_account_pubkey,
526            nonce_authority_pubkey,
527        );
528        instructions.insert(0, nonce_ix);
529        Self::new(&instructions, payer)
530    }

559    /// Compute the blake3 hash of a raw transaction message.
560    #[cfg(all(not(target_os = "solana"), feature = "blake3"))]
561    pub fn hash_raw_message(message_bytes: &[u8]) -> Hash {
562        use {blake3::traits::digest::Digest, solana_hash::HASH_BYTES};
563        let mut hasher = blake3::Hasher::new();
564        hasher.update(b"solana-tx-message-v1");
565        hasher.update(message_bytes);
566        let hash_bytes: [u8; HASH_BYTES] = hasher.finalize().into();
567        hash_bytes.into()
568    }

601    /// Returns true if the account at the specified index is an account input
602    /// to some program instruction in this message.
603    pub fn is_instruction_account(&self, key_index: usize) -> bool {
604        if let Ok(key_index) = u8::try_from(key_index) {
605            self.instructions
606                .iter()
607                .any(|ix| ix.accounts.contains(&key_index))
608        } else {
609            false
610        }
611    }
612
613    pub fn is_key_called_as_program(&self, key_index: usize) -> bool {
614        if let Ok(key_index) = u8::try_from(key_index) {
615            self.instructions
616                .iter()
617                .any(|ix| ix.program_id_index == key_index)
618        } else {
619            false
620        }
621    }
622
623    #[deprecated(
624        since = "2.0.0",
625        note = "Please use `is_key_called_as_program` and `is_instruction_account` directly"
626    )]
627    pub fn is_non_loader_key(&self, key_index: usize) -> bool {
628        !self.is_key_called_as_program(key_index) || self.is_instruction_account(key_index)
629    }
630
631    pub fn program_position(&self, index: usize) -> Option<usize> {
632        let program_ids = self.program_ids();
633        program_ids
634            .iter()
635            .position(|&&pubkey| pubkey == self.account_keys[index])
636    }
637
638    pub fn maybe_executable(&self, i: usize) -> bool {
639        self.program_position(i).is_some()
640    }
641
642    pub fn demote_program_id(&self, i: usize) -> bool {
643        self.is_key_called_as_program(i) && !self.is_upgradeable_loader_present()
644    }
645
646    /// Returns true if the account at the specified index was requested to be
647    /// writable. This method should not be used directly.
648    pub(super) fn is_writable_index(&self, i: usize) -> bool {
649        i < (self.header.num_required_signatures - self.header.num_readonly_signed_accounts)
650            as usize
651            || (i >= self.header.num_required_signatures as usize
652                && i < self.account_keys.len()
653                    - self.header.num_readonly_unsigned_accounts as usize)
654    }
655
656    /// Returns true if the account at the specified index is writable by the
657    /// instructions in this message. Since the dynamic set of reserved accounts
658    /// isn't used here to demote write locks, this shouldn't be used in the
659    /// runtime.
660    #[deprecated(since = "2.0.0", note = "Please use `is_maybe_writable` instead")]
661    #[allow(deprecated)]
662    pub fn is_writable(&self, i: usize) -> bool {
663        (self.is_writable_index(i))
664            && !is_builtin_key_or_sysvar(&self.account_keys[i])
665            && !self.demote_program_id(i)
666    }
667
668    /// Returns true if the account at the specified index is writable by the
669    /// instructions in this message. The `reserved_account_keys` param has been
670    /// optional to allow clients to approximate writability without requiring
671    /// fetching the latest set of reserved account keys. If this method is
672    /// called by the runtime, the latest set of reserved account keys must be
673    /// passed.
674    pub fn is_maybe_writable(
675        &self,
676        i: usize,
677        reserved_account_keys: Option<&HashSet<Pubkey>>,
678    ) -> bool {
679        (self.is_writable_index(i))
680            && !self.is_account_maybe_reserved(i, reserved_account_keys)
681            && !self.demote_program_id(i)
682    }
683
684    /// Returns true if the account at the specified index is in the optional
685    /// reserved account keys set.
686    fn is_account_maybe_reserved(
687        &self,
688        key_index: usize,
689        reserved_account_keys: Option<&HashSet<Pubkey>>,
690    ) -> bool {
691        let mut is_maybe_reserved = false;
692        if let Some(reserved_account_keys) = reserved_account_keys {
693            if let Some(key) = self.account_keys.get(key_index) {
694                is_maybe_reserved = reserved_account_keys.contains(key);
695            }
696        }
697        is_maybe_reserved
698    }
699
700    pub fn is_signer(&self, i: usize) -> bool {
701        i < self.header.num_required_signatures as usize
702    }
703
704    pub fn signer_keys(&self) -> Vec<&Pubkey> {
705        // Clamp in case we're working on un-`sanitize()`ed input
706        let last_key = self
707            .account_keys
708            .len()
709            .min(self.header.num_required_signatures as usize);
710        self.account_keys[..last_key].iter().collect()
711    }
712
713    /// Returns `true` if `account_keys` has any duplicate keys.
714    pub fn has_duplicates(&self) -> bool {
715        // Note: This is an O(n^2) algorithm, but requires no heap allocations. The benchmark
716        // `bench_has_duplicates` in benches/message_processor.rs shows that this implementation is
717        // ~50 times faster than using HashSet for very short slices.
718        for i in 1..self.account_keys.len() {
719            #[allow(clippy::arithmetic_side_effects)]
720            if self.account_keys[i..].contains(&self.account_keys[i - 1]) {
721                return true;
722            }
723        }
724        false
725    }
726
727    /// Returns `true` if any account is the BPF upgradeable loader.
728    pub fn is_upgradeable_loader_present(&self) -> bool {
729        self.account_keys
730            .iter()
731            .any(|&key| key == bpf_loader_upgradeable::id())
732    }
