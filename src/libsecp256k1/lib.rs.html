120    pub fn parse_slice(p: &[u8], format: Option<PublicKeyFormat>) -> Result<PublicKey, Error> {
121        let format = match (p.len(), format) {
122            (util::FULL_PUBLIC_KEY_SIZE, None)
123            | (util::FULL_PUBLIC_KEY_SIZE, Some(PublicKeyFormat::Full)) => PublicKeyFormat::Full,
124            (util::COMPRESSED_PUBLIC_KEY_SIZE, None)
125            | (util::COMPRESSED_PUBLIC_KEY_SIZE, Some(PublicKeyFormat::Compressed)) => {
126                PublicKeyFormat::Compressed
127            }
128            (util::RAW_PUBLIC_KEY_SIZE, None)
129            | (util::RAW_PUBLIC_KEY_SIZE, Some(PublicKeyFormat::Raw)) => PublicKeyFormat::Raw,
130            _ => return Err(Error::InvalidInputLength),
131        };
132
133        match format {
134            PublicKeyFormat::Full => {
135                let mut a = [0; util::FULL_PUBLIC_KEY_SIZE];
136                a.copy_from_slice(p);
137                Self::parse(&a)
138            }
139            PublicKeyFormat::Raw => {
140                use util::TAG_PUBKEY_FULL;
141
142                let mut a = [0; util::FULL_PUBLIC_KEY_SIZE];
143                a[0] = TAG_PUBKEY_FULL;
144                a[1..].copy_from_slice(p);
145                Self::parse(&a)
146            }
147            PublicKeyFormat::Compressed => {
148                let mut a = [0; util::COMPRESSED_PUBLIC_KEY_SIZE];
149                a.copy_from_slice(p);
150                Self::parse_compressed(&a)
151            }
152        }
153    }
154
155    pub fn parse(p: &[u8; util::FULL_PUBLIC_KEY_SIZE]) -> Result<PublicKey, Error> {
156        use util::{TAG_PUBKEY_FULL, TAG_PUBKEY_HYBRID_EVEN, TAG_PUBKEY_HYBRID_ODD};
157
158        if !(p[0] == TAG_PUBKEY_FULL
159            || p[0] == TAG_PUBKEY_HYBRID_EVEN
160            || p[0] == TAG_PUBKEY_HYBRID_ODD)
161        {
162            return Err(Error::InvalidPublicKey);
163        }
164        let mut x = Field::default();
165        let mut y = Field::default();
166        if !x.set_b32(array_ref!(p, 1, 32)) {
167            return Err(Error::InvalidPublicKey);
168        }
169        if !y.set_b32(array_ref!(p, 33, 32)) {
170            return Err(Error::InvalidPublicKey);
171        }
172        let mut elem = Affine::default();
173        elem.set_xy(&x, &y);
174        if (p[0] == TAG_PUBKEY_HYBRID_EVEN || p[0] == TAG_PUBKEY_HYBRID_ODD)
175            && (y.is_odd() != (p[0] == TAG_PUBKEY_HYBRID_ODD))
176        {
177            return Err(Error::InvalidPublicKey);
178        }
179        if elem.is_infinity() {
180            return Err(Error::InvalidPublicKey);
181        }
182        if elem.is_valid_var() {
183            Ok(PublicKey(elem))
184        } else {
185            Err(Error::InvalidPublicKey)
186        }
187    }
188
189    pub fn parse_compressed(
190        p: &[u8; util::COMPRESSED_PUBLIC_KEY_SIZE],
191    ) -> Result<PublicKey, Error> {
192        use util::{TAG_PUBKEY_EVEN, TAG_PUBKEY_ODD};
193
194        if !(p[0] == TAG_PUBKEY_EVEN || p[0] == TAG_PUBKEY_ODD) {
195            return Err(Error::InvalidPublicKey);
196        }
197        let mut x = Field::default();
198        if !x.set_b32(array_ref!(p, 1, 32)) {
199            return Err(Error::InvalidPublicKey);
200        }
201        let mut elem = Affine::default();
202        elem.set_xo_var(&x, p[0] == TAG_PUBKEY_ODD);
203        if elem.is_infinity() {
204            return Err(Error::InvalidPublicKey);
205        }
206        if elem.is_valid_var() {
207            Ok(PublicKey(elem))
208        } else {
209            Err(Error::InvalidPublicKey)
210        }
211    }

213    pub fn serialize(&self) -> [u8; util::FULL_PUBLIC_KEY_SIZE] {
214        use util::TAG_PUBKEY_FULL;
215
216        debug_assert!(!self.0.is_infinity());
217
218        let mut ret = [0u8; 65];
219        let mut elem = self.0;
220
221        elem.x.normalize_var();
222        elem.y.normalize_var();
223        elem.x.fill_b32(array_mut_ref!(ret, 1, 32));
224        elem.y.fill_b32(array_mut_ref!(ret, 33, 32));
225        ret[0] = TAG_PUBKEY_FULL;
226
227        ret
228    }
229
230    pub fn serialize_compressed(&self) -> [u8; util::COMPRESSED_PUBLIC_KEY_SIZE] {
231        use util::{TAG_PUBKEY_EVEN, TAG_PUBKEY_ODD};
232
233        debug_assert!(!self.0.is_infinity());
234
235        let mut ret = [0u8; 33];
236        let mut elem = self.0;
237
238        elem.x.normalize_var();
239        elem.y.normalize_var();
240        elem.x.fill_b32(array_mut_ref!(ret, 1, 32));
241        ret[0] = if elem.y.is_odd() {
242            TAG_PUBKEY_ODD
243        } else {
244            TAG_PUBKEY_EVEN
245        };
246
247        ret
248    }

250    pub fn tweak_add_assign_with_context(
251        &mut self,
252        tweak: &SecretKey,
253        context: &ECMultContext,
254    ) -> Result<(), Error> {
255        let mut r = Jacobian::default();
256        let a = Jacobian::from_ge(&self.0);
257        let one = Scalar::from_int(1);
258        context.ecmult(&mut r, &a, &one, &tweak.0);
259
260        if r.is_infinity() {
261            return Err(Error::TweakOutOfRange);
262        }
263
264        self.0.set_gej(&r);
265        Ok(())
266    }

273    pub fn tweak_mul_assign_with_context(
274        &mut self,
275        tweak: &SecretKey,
276        context: &ECMultContext,
277    ) -> Result<(), Error> {
278        if tweak.0.is_zero() {
279            return Err(Error::TweakOutOfRange);
280        }
281
282        let mut r = Jacobian::default();
283        let zero = Scalar::from_int(0);
284        let pt = Jacobian::from_ge(&self.0);
285        context.ecmult(&mut r, &pt, &tweak.0, &zero);
286
287        self.0.set_gej(&r);
288        Ok(())
289    }
290
291    #[cfg(any(feature = "static-context", feature = "lazy-static-context"))]
292    pub fn tweak_mul_assign(&mut self, tweak: &SecretKey) -> Result<(), Error> {
293        self.tweak_mul_assign_with_context(tweak, &ECMULT_CONTEXT)
294    }

296    pub fn combine(keys: &[PublicKey]) -> Result<Self, Error> {
297        let mut qj = Jacobian::default();
298        qj.set_infinity();
299
300        for key in keys {
301            qj = qj.add_ge(&key.0);
302        }
303
304        if qj.is_infinity() {
305            return Err(Error::InvalidPublicKey);
306        }
307
308        let q = Affine::from_gej(&qj);
309        Ok(PublicKey(q))
310    }

336#[cfg(feature = "std")]
337impl<'de> de::Visitor<'de> for PublicKeyVisitor {
338    type Value = PublicKey;
339
340    fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
341        formatter
342            .write_str("a bytestring of either 33 (compressed), 64 (raw), or 65 bytes in length")
343    }
344
345    fn visit_str<E>(self, value: &str) -> Result<Self::Value, E>
346    where
347        E: de::Error,
348    {
349        let value: &[u8] = &base64::decode(value).unwrap();
350        let key_format = match value.len() {
351            33 => PublicKeyFormat::Compressed,
352            64 => PublicKeyFormat::Raw,
353            65 => PublicKeyFormat::Full,
354            _ => return Err(E::custom(Error::InvalidInputLength)),
355        };
356        PublicKey::parse_slice(value, Some(key_format))
357            .map_err(|_e| E::custom(Error::InvalidPublicKey))
358    }
359}

375impl SecretKey {
376    pub fn parse(p: &[u8; util::SECRET_KEY_SIZE]) -> Result<SecretKey, Error> {
377        let mut elem = Scalar::default();
378        if !bool::from(elem.set_b32(p)) {
379            Self::try_from(elem)
380        } else {
381            Err(Error::InvalidSecretKey)
382        }
383    }
384
385    pub fn parse_slice(p: &[u8]) -> Result<SecretKey, Error> {
386        if p.len() != util::SECRET_KEY_SIZE {
387            return Err(Error::InvalidInputLength);
388        }
389
390        let mut a = [0; 32];
391        a.copy_from_slice(p);
392        Self::parse(&a)
393    }
394
395    pub fn random<R: Rng>(rng: &mut R) -> SecretKey {
396        loop {
397            let mut ret = [0u8; util::SECRET_KEY_SIZE];
398            rng.fill_bytes(&mut ret);
399
400            if let Ok(key) = Self::parse(&ret) {
401                return key;
402            }
403        }
404    }

408    }

411        let v = self.0 + tweak.0;
412        if v.is_zero() {
413            return Err(Error::TweakOutOfRange);

416        Ok(())

473impl Signature {
474    /// Parse an possibly overflowing signature.
475    ///
476    /// A SECP256K1 signature is usually required to be within 0 and curve
477    /// order. This function, however, allows signatures larger than curve order
478    /// by taking the signature and minus curve order.
479    ///
480    /// Note that while this function is technically safe, it is non-standard,
481    /// meaning you will have compatibility issues if you also use other
482    /// SECP256K1 libraries. It's not recommended to use this function. Please
483    /// use `parse_standard` instead.
484    pub fn parse_overflowing(p: &[u8; util::SIGNATURE_SIZE]) -> Signature {
485        let mut r = Scalar::default();
486        let mut s = Scalar::default();
487
488        // Okay for signature to overflow
489        let _ = r.set_b32(array_ref!(p, 0, 32));
490        let _ = s.set_b32(array_ref!(p, 32, 32));
491
492        Signature { r, s }
493    }
494
495    /// Parse a standard SECP256K1 signature. The signature is required to be
496    /// within 0 and curve order. Returns error if it overflows.
497    pub fn parse_standard(p: &[u8; util::SIGNATURE_SIZE]) -> Result<Signature, Error> {
498        let mut r = Scalar::default();
499        let mut s = Scalar::default();
500
501        // It's okay for the signature to overflow here, it's checked below.
502        let overflowed_r = r.set_b32(array_ref!(p, 0, 32));
503        let overflowed_s = s.set_b32(array_ref!(p, 32, 32));
504
505        if bool::from(overflowed_r | overflowed_s) {
506            return Err(Error::InvalidSignature);
507        }
508
509        Ok(Signature { r, s })
510    }
511
512    /// Parse a possibly overflowing signature slice. See also
513    /// `parse_overflowing`.
514    ///
515    /// It's not recommended to use this function. Please use
516    /// `parse_standard_slice` instead.
517    pub fn parse_overflowing_slice(p: &[u8]) -> Result<Signature, Error> {
518        if p.len() != util::SIGNATURE_SIZE {
519            return Err(Error::InvalidInputLength);
520        }
521
522        let mut a = [0; util::SIGNATURE_SIZE];
523        a.copy_from_slice(p);
524        Ok(Self::parse_overflowing(&a))
525    }
526
527    /// Parse a standard signature slice. See also `parse_standard`.
528    pub fn parse_standard_slice(p: &[u8]) -> Result<Signature, Error> {
529        if p.len() != util::SIGNATURE_SIZE {
530            return Err(Error::InvalidInputLength);
531        }
532
533        let mut a = [0; util::SIGNATURE_SIZE];
534        a.copy_from_slice(p);
535        Ok(Self::parse_standard(&a)?)
536    }

538    /// Parse a DER-encoded byte slice to a signature.
539    pub fn parse_der(p: &[u8]) -> Result<Signature, Error> {
540        let mut decoder = Decoder::new(p);
541
542        decoder.read_constructed_sequence()?;
543        let rlen = decoder.read_len()?;
544
545        if rlen != decoder.remaining_len() {
546            return Err(Error::InvalidSignature);
547        }
548
549        let r = decoder.read_integer()?;
550        let s = decoder.read_integer()?;
551
552        if decoder.remaining_len() != 0 {
553            return Err(Error::InvalidSignature);
554        }
555
556        Ok(Signature { r, s })
557    }
558
559    /// Converts a "lax DER"-encoded byte slice to a signature. This is basically
560    /// only useful for validating signatures in the Bitcoin blockchain from before
561    /// 2016. It should never be used in new applications. This library does not
562    /// support serializing to this "format"
563    pub fn parse_der_lax(p: &[u8]) -> Result<Signature, Error> {
564        let mut decoder = Decoder::new(p);
565
566        decoder.read_constructed_sequence()?;
567        decoder.read_seq_len_lax()?;
568
569        let r = decoder.read_integer_lax()?;
570        let s = decoder.read_integer_lax()?;
571
572        Ok(Signature { r, s })
573    }
574
575    /// Normalizes a signature to a "low S" form. In ECDSA, signatures are
576    /// of the form (r, s) where r and s are numbers lying in some finite
577    /// field. The verification equation will pass for (r, s) iff it passes
578    /// for (r, -s), so it is possible to ``modify'' signatures in transit
579    /// by flipping the sign of s. This does not constitute a forgery since
580    /// the signed message still cannot be changed, but for some applications,
581    /// changing even the signature itself can be a problem. Such applications
582    /// require a "strong signature". It is believed that ECDSA is a strong
583    /// signature except for this ambiguity in the sign of s, so to accommodate
584    /// these applications libsecp256k1 will only accept signatures for which
585    /// s is in the lower half of the field range. This eliminates the
586    /// ambiguity.
587    ///
588    /// However, for some systems, signatures with high s-values are considered
589    /// valid. (For example, parsing the historic Bitcoin blockchain requires
590    /// this.) For these applications we provide this normalization function,
591    /// which ensures that the s value lies in the lower half of its range.
592    pub fn normalize_s(&mut self) {
593        if self.s.is_high() {
594            self.s = -self.s;
595        }
596    }
597
598    /// Serialize a signature to a standard byte representation. This is the
599    /// reverse of `parse_standard`.
600    pub fn serialize(&self) -> [u8; util::SIGNATURE_SIZE] {
601        let mut ret = [0u8; 64];
602        self.r.fill_b32(array_mut_ref!(ret, 0, 32));
603        self.s.fill_b32(array_mut_ref!(ret, 32, 32));
604        ret
605    }
606
607    /// Serialize a signature to a DER encoding. This is the reverse of
608    /// `parse_der`.
609    pub fn serialize_der(&self) -> SignatureArray {
610        fn fill_scalar_with_leading_zero(scalar: &Scalar) -> [u8; 33] {
611            let mut ret = [0u8; 33];
612            scalar.fill_b32(array_mut_ref!(ret, 1, 32));
613            ret
614        }
615
616        let r_full = fill_scalar_with_leading_zero(&self.r);
617        let s_full = fill_scalar_with_leading_zero(&self.s);
618
619        fn integer_slice(full: &[u8; 33]) -> &[u8] {
620            let mut len = 33;
621            while len > 1 && full[full.len() - len] == 0 && full[full.len() - len + 1] < 0x80 {
622                len -= 1;
623            }
624            &full[(full.len() - len)..]
625        }
626
627        let r = integer_slice(&r_full);
628        let s = integer_slice(&s_full);
629
630        let mut ret = SignatureArray::new(6 + r.len() + s.len());
631        {
632            let l = ret.as_mut();
633            l[0] = 0x30;
634            l[1] = 4 + r.len() as u8 + s.len() as u8;
635            l[2] = 0x02;
636            l[3] = r.len() as u8;
637            l[4..(4 + r.len())].copy_from_slice(r);
638            l[4 + r.len()] = 0x02;
639            l[5 + r.len()] = s.len() as u8;
640            l[(6 + r.len())..(6 + r.len() + s.len())].copy_from_slice(s);
641        }
642
643        ret
644    }
645}

656
657    pub fn parse_slice(p: &[u8]) -> Result<Message, Error> {
658        if p.len() != util::MESSAGE_SIZE {
659            return Err(Error::InvalidInputLength);
660        }
661
662        let mut a = [0; util::MESSAGE_SIZE];
663        a.copy_from_slice(p);
664        Ok(Self::parse(&a))
665    }
666
667    pub fn serialize(&self) -> [u8; util::MESSAGE_SIZE] {
668        self.0.b32()
669    }
670}

772#[cfg(feature = "hmac")]
773/// Sign a message using the secret key, with the given context.
774pub fn sign_with_context(
775    message: &Message,
776    seckey: &SecretKey,
777    context: &ECMultGenContext,
778) -> (Signature, RecoveryId) {
779    let seckey_b32 = seckey.0.b32();
780    let message_b32 = message.0.b32();
781
782    let mut drbg = HmacDRBG::<Sha256>::new(&seckey_b32, &message_b32, &[]);
783    let mut nonce = Scalar::default();
784    let mut overflow;
785
786    let result;
787    loop {
788        let generated = drbg.generate::<U32>(None);
789        overflow = bool::from(nonce.set_b32(array_ref!(generated, 0, 32)));
790
791        if !overflow && !nonce.is_zero() {
792            if let Ok(val) = context.sign_raw(&seckey.0, &message.0, &nonce) {
793                result = val;
794                break;
795            }
796        }
797    }
798
799    #[allow(unused_assignments)]
800    {
801        nonce = Scalar::default();
802    }
803    let (sigr, sigs, recid) = result;
804
805    (Signature { r: sigr, s: sigs }, RecoveryId(recid))
806}
807
808#[cfg(all(
809    feature = "hmac",
810    any(feature = "static-context", feature = "lazy-static-context")
811))]
812/// Sign a message using the secret key.
813pub fn sign(message: &Message, seckey: &SecretKey) -> (Signature, RecoveryId) {
814    sign_with_context(message, seckey, &ECMULT_GEN_CONTEXT)
815}
