1use core::convert::TryFrom;
2
3use proc_macro2::{Span, TokenStream as TokenStream2};
4use quote::quote;
5use syn::{Fields, Ident, ItemEnum, WhereClause};
6
7use crate::attribute_helpers::contains_skip;

9pub fn enum_ser(input: &ItemEnum, cratename: Ident) -> syn::Result<TokenStream2> {
10    let name = &input.ident;
11    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
12    let mut where_clause = where_clause.map_or_else(
13        || WhereClause {
14            where_token: Default::default(),
15            predicates: Default::default(),
16        },
17        Clone::clone,
18    );
19    let mut variant_idx_body = TokenStream2::new();
20    let mut fields_body = TokenStream2::new();
21    for (variant_idx, variant) in input.variants.iter().enumerate() {
22        let variant_idx = u8::try_from(variant_idx).expect("up to 256 enum variants are supported");
23        let variant_ident = &variant.ident;
24        let mut variant_header = TokenStream2::new();
25        let mut variant_body = TokenStream2::new();
26        match &variant.fields {
27            Fields::Named(fields) => {
28                for field in &fields.named {
29                    let field_name = field.ident.as_ref().unwrap();
30                    if contains_skip(&field.attrs) {
31                        variant_header.extend(quote! { _#field_name, });
32                        continue;
33                    } else {
34                        let field_type = &field.ty;
35                        where_clause.predicates.push(
36                            syn::parse2(quote! {
37                                #field_type: #cratename::ser::BorshSerialize
38                            })
39                            .unwrap(),
40                        );
41                        variant_header.extend(quote! { #field_name, });
42                    }
43                    variant_body.extend(quote! {
44                         #cratename::BorshSerialize::serialize(#field_name, writer)?;
45                    })
46                }
47                variant_header = quote! { { #variant_header }};
48                variant_idx_body.extend(quote!(
49                    #name::#variant_ident { .. } => #variant_idx,
50                ));
51            }
52            Fields::Unnamed(fields) => {
53                for (field_idx, field) in fields.unnamed.iter().enumerate() {
54                    let field_idx =
55                        u32::try_from(field_idx).expect("up to 2^32 fields are supported");
56                    if contains_skip(&field.attrs) {
57                        let field_ident =
58                            Ident::new(format!("_id{}", field_idx).as_str(), Span::call_site());
59                        variant_header.extend(quote! { #field_ident, });
60                        continue;
61                    } else {
62                        let field_type = &field.ty;
63                        where_clause.predicates.push(
64                            syn::parse2(quote! {
65                                #field_type: #cratename::ser::BorshSerialize
66                            })
67                            .unwrap(),
68                        );
69
70                        let field_ident =
71                            Ident::new(format!("id{}", field_idx).as_str(), Span::call_site());
72                        variant_header.extend(quote! { #field_ident, });
73                        variant_body.extend(quote! {
74                            #cratename::BorshSerialize::serialize(#field_ident, writer)?;
75                        })
76                    }
77                }
78                variant_header = quote! { ( #variant_header )};
79                variant_idx_body.extend(quote!(
80                    #name::#variant_ident(..) => #variant_idx,
81                ));
82            }
83            Fields::Unit => {
84                variant_idx_body.extend(quote!(
85                    #name::#variant_ident => #variant_idx,
86                ));
87            }
88        }
89        fields_body.extend(quote!(
90            #name::#variant_ident #variant_header => {
91                #variant_body
92            }
93        ))
94    }
95    Ok(quote! {
96        impl #impl_generics #cratename::ser::BorshSerialize for #name #ty_generics #where_clause {
97            fn serialize<W: #cratename::maybestd::io::Write>(&self, writer: &mut W) -> ::core::result::Result<(), #cratename::maybestd::io::Error> {
98                let variant_idx: u8 = match self {
99                    #variant_idx_body
100                };
101                writer.write_all(&variant_idx.to_le_bytes())?;
102
103                match self {
104                    #fields_body
105                }
106                Ok(())
107            }
108        }
109    })
110}
