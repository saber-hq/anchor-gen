1//! Defines the instructions and account state used to store a program's
2//! IDL on-chain at a canonical account address, which can be derived as a
3//! function of nothing other than the program's ID.
4//!
5//! It can be upgraded in a way similar to a BPF upgradeable program. That is,
6//! one may invoke the `IdlInstruction::CreateBuffer` instruction to create
7//! a buffer, `IdlInstruction::Write` to write a new IDL into it, and then
8//! `IdlInstruction::SetBuffer` to copy the IDL into the program's canonical
9//! IDL account. In order to perform this upgrade, the buffer's `authority`
10//! must match the canonical IDL account's authority.
11//!
12//! Because the IDL can be larger than the max transaction size, the transaction
13//! must be broken up into several pieces and stored into the IDL account with
14//! multiple transactions via the `Write` instruction to continuously append to
15//! the account's IDL data buffer.
16//!
17//! Note that IDL account instructions are automatically inserted into all
18//! Anchor programs. To remove them, one can use the `no-idl` feature.
19
20use crate::prelude::*;
21
22// The first 8 bytes of an instruction to create or modify the IDL account. This
23// instruction is defined outside the main program's instruction enum, so that
24// the enum variant tags can align with function source order.
25//
26// Sha256(anchor:idl)[..8];
27pub const IDL_IX_TAG: u64 = 0x0a69e9a778bcf440;
28pub const IDL_IX_TAG_LE: &[u8] = IDL_IX_TAG.to_le_bytes().as_slice();
29
30// The Pubkey that is stored as the 'authority' on the IdlAccount when the authority
31// is "erased".
32pub const ERASED_AUTHORITY: Pubkey = Pubkey::new_from_array([0u8; 32]);
33
34#[derive(AnchorSerialize, AnchorDeserialize)]
35pub enum IdlInstruction {
36    // One time initializer for creating the program's idl account.
37    Create { data_len: u64 },
38    // Creates a new IDL account buffer. Can be called several times.
39    CreateBuffer,
40    // Appends the given data to the end of the idl account buffer.
41    Write { data: Vec<u8> },
42    // Sets a new data buffer for the IdlAccount.
43    SetBuffer,
44    // Sets a new authority on the IdlAccount.
45    SetAuthority { new_authority: Pubkey },
46    Close,
47    // Increases account size for accounts that need over 10kb.
48    Resize { data_len: u64 },
49}
50
51// The account holding a program's IDL. This is stored on chain so that clients
52// can fetch it and generate a client with nothing but a program's ID.
53//
54// Note: we use the same account for the "write buffer", similar to the
55//       bpf upgradeable loader's mechanism.
56//
57// TODO: IdlAccount exists here only because it's needed by the CLI, the IDL
58// itself uses an IdlAccount defined inside the program itself, see program/idl.rs.
59// Ideally it would be deleted and a better solution for sharing the type with CLI
60// could be found.
61#[account("internal")]
62#[derive(Debug)]
63pub struct IdlAccount {
64    // Address that can modify the IDL.
65    pub authority: Pubkey,
66    // Length of compressed idl bytes.
67    pub data_len: u32,
68    // Followed by compressed idl bytes.
69}
70
71impl IdlAccount {
72    pub fn address(program_id: &Pubkey) -> Pubkey {
73        let program_signer = Pubkey::find_program_address(&[], program_id).0;
74        Pubkey::create_with_seed(&program_signer, IdlAccount::seed(), program_id)
75            .expect("Seed is always valid")
76    }
77    pub fn seed() -> &'static str {
78        "anchor:idl"
79    }
80}
