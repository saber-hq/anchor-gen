12/// Type validating that the account is a sysvar and deserializing it.
13///
14/// If possible, sysvars should not be used via accounts
15/// but by using the [`get`](https://docs.rs/solana-program/latest/solana_program/sysvar/trait.Sysvar.html#method.get)
16/// function on the desired sysvar. This is because using `get`
17/// does not run the risk of Anchor having a bug in its `Sysvar` type
18/// and using `get` also decreases tx size, making space for other
19/// accounts that cannot be requested via syscall.
20///
21/// # Example
22/// ```ignore
23/// // OK - via account in the account validation struct
24/// #[derive(Accounts)]
25/// pub struct Example<'info> {
26///     pub clock: Sysvar<'info, Clock>
27/// }
28/// // BETTER - via syscall in the instruction function
29/// fn better(ctx: Context<Better>) -> Result<()> {
30///     let clock = Clock::get()?;
31/// }
32/// ```
33pub struct Sysvar<'info, T: solana_program::sysvar::Sysvar> {
34    info: &'info AccountInfo<'info>,
35    account: T,
36}
37
38impl<T: solana_program::sysvar::Sysvar + fmt::Debug> fmt::Debug for Sysvar<'_, T> {
39    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
40        f.debug_struct("Sysvar")
41            .field("info", &self.info)
42            .field("account", &self.account)
43            .finish()
44    }
45}
46
47impl<'info, T: solana_program::sysvar::Sysvar> Sysvar<'info, T> {
48    pub fn from_account_info(acc_info: &'info AccountInfo<'info>) -> Result<Sysvar<'info, T>> {
49        match T::from_account_info(acc_info) {
50            Ok(val) => Ok(Sysvar {
51                info: acc_info,
52                account: val,
53            }),
54            Err(_) => Err(ErrorCode::AccountSysvarMismatch.into()),
55        }
56    }
57}

59impl<T: solana_program::sysvar::Sysvar> Clone for Sysvar<'_, T> {

64        }
65    }
66}

68impl<'info, B, T: solana_program::sysvar::Sysvar> Accounts<'info, B> for Sysvar<'info, T> {
69    fn try_accounts(
70        _program_id: &Pubkey,
71        accounts: &mut &'info [AccountInfo<'info>],
72        _ix_data: &[u8],
73        _bumps: &mut B,
74        _reallocs: &mut BTreeSet<Pubkey>,
75    ) -> Result<Self> {
76        if accounts.is_empty() {
77            return Err(ErrorCode::AccountNotEnoughKeys.into());
78        }
79        let account = &accounts[0];
80        *accounts = &accounts[1..];
81        Sysvar::from_account_info(account)
82    }
83}
