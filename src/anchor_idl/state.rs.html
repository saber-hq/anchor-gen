12/// Generates an account state struct.
13pub fn generate_account(
14    defs: &[IdlTypeDef],
15    account_name: &str,
16    fields: &[IdlField],
17    opts: StructOpts,
18) -> TokenStream {
19    let props = get_field_list_properties(defs, fields);
20
21    let derive_copy = if props.can_copy && !opts.zero_copy {
22        quote! {
23            #[derive(Copy)]
24        }
25    } else {
26        quote! {}
27    };
28    let derive_default = if props.can_derive_default {
29        quote! {
30            #[derive(Default)]
31        }
32    } else {
33        quote! {}
34    };
35    let derive_account = if opts.zero_copy {
36        let repr = if opts.packed {
37            quote! {
38                #[repr(packed)]
39            }
40        } else {
41            quote! {
42                #[repr(C)]
43            }
44        };
45        quote! {
46            #[account(zero_copy(unsafe))]
47            #repr
48        }
49    } else {
50        quote! {
51            #[account]
52        }
53    };
54
55    let doc = format!(" Account: {}", account_name);
56    let struct_name = format_ident!("{}", account_name);
57    let fields_rendered = generate_struct_fields_from_slice(fields);
58    quote! {
59        #derive_account
60        #[doc = #doc]
61        #derive_copy
62        #derive_default
63        pub struct #struct_name {
64            #fields_rendered
65        }
66    }
67}

69/// Generates account state structs.
70pub fn generate_accounts(
71    typedefs: &[IdlTypeDef],
72    account_defs: &[IdlAccount],
73    struct_opts: &BTreeMap<String, StructOpts>,
74) -> TokenStream {
75    let defined = account_defs
76        .iter()
77        .map(|account| {
78            typedefs
79                .iter()
80                .find(|type_def| type_def.name == account.name)
81                .unwrap()
82        })
83        .map(|def| match &def.ty {
84            anchor_lang_idl_spec::IdlTypeDefTy::Struct { fields } => {
85                let opts = struct_opts.get(&def.name).copied().unwrap_or_default();
86                generate_account(
87                    typedefs,
88                    &def.name,
89                    get_idl_defined_fields_as_slice(fields),
90                    opts,
91                )
92            }
93            anchor_lang_idl_spec::IdlTypeDefTy::Enum { .. } => {
94                panic!("unexpected enum account");
95            }
96            anchor_lang_idl_spec::IdlTypeDefTy::Type { alias: _ } => {
97                panic!("unexpected type account")
98            }
99        });
100    quote! {
101        #(#defined)*
102    }
103}
