6/// Generates a single instruction handler.
7pub fn generate_ix_handler(ix: &IdlInstruction) -> TokenStream {
8    let ix_name = format_ident!("{}", ix.name.to_snake_case());
9    let accounts_name = format_ident!("{}", ix.name.to_pascal_case());
10
11    let args = ix
12        .args
13        .iter()
14        .map(|arg| {
15            let name = format_ident!("_{}", arg.name.to_snake_case());
16            let type_name = crate::ty_to_rust_type(&arg.ty);
17            let stream: proc_macro2::TokenStream = type_name.parse().unwrap();
18            quote! {
19                #name: #stream
20            }
21        })
22        .collect::<Vec<_>>();
23
24    if cfg!(feature = "compat-program-result") {
25        quote! {
26            pub fn #ix_name(
27                _ctx: Context<#accounts_name>,
28                #(#args),*
29            ) -> ProgramResult {
30                unimplemented!("This program is a wrapper for CPI.")
31            }
32        }
33    } else {
34        quote! {
35            pub fn #ix_name(
36                _ctx: Context<#accounts_name>,
37                #(#args),*
38            ) -> Result<()> {
39                unimplemented!("This program is a wrapper for CPI.")
40            }
41        }
42    }
43}

45/// Generates instruction context structs.
46pub fn generate_ix_structs(ixs: &[IdlInstruction]) -> TokenStream {
47    let defs = ixs.iter().map(|ix| {
48        let accounts_name = format_ident!("{}", ix.name.to_pascal_case());
49
50        let (all_structs, all_fields) =
51            crate::generate_account_fields(&ix.name.to_pascal_case(), &ix.accounts);
52
53        let struct_def = if ix.accounts.is_empty() {
54            quote! {
55                pub struct #accounts_name {
56                    #all_fields
57                }
58            }
59        } else {
60            quote! {
61                pub struct #accounts_name<'info> {
62                    #all_fields
63                }
64            }
65        };
66
67        quote! {
68            #all_structs
69
70            #[derive(Accounts)]
71            #struct_def
72        }
73    });
74    quote! {
75        #(#defs)*
76    }
77}

79/// Generates all instruction handlers.
80pub fn generate_ix_handlers(ixs: &[IdlInstruction]) -> TokenStream {
81    let streams = ixs.iter().map(generate_ix_handler);
82    quote! {
83        #(#streams)*
84    }
