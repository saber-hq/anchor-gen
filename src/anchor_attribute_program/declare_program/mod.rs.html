18pub struct DeclareProgram {
19    name: syn::Ident,
20    idl: Idl,
21}
22
23impl Parse for DeclareProgram {
24    fn parse(input: ParseStream) -> syn::Result<Self> {
25        let name = input.parse()?;
26        let idl = get_idl(&name).map_err(|e| syn::Error::new(name.span(), e))?;
27        Ok(Self { name, idl })
28    }
29}
30
31impl ToTokens for DeclareProgram {
32    fn to_tokens(&self, tokens: &mut proc_macro2::TokenStream) {
33        let program = gen_program(&self.idl, &self.name);
34        tokens.extend(program)
35    }
36}

38fn get_idl(name: &syn::Ident) -> anyhow::Result<Idl> {
39    env::var("CARGO_MANIFEST_DIR")
40        .map(PathBuf::from)
41        .map_err(|e| anyhow!("Failed to get environment variable `CARGO_MANIFEST_DIR`: {e}"))?
42        .ancestors()
43        .find_map(|ancestor| {
44            let idl_dir = ancestor.join("idls");
45            idl_dir.exists().then_some(idl_dir)
46        })
47        .ok_or_else(|| anyhow!("`idls` directory not found"))
48        .map(|idl_dir| idl_dir.join(name.to_string()).with_extension("json"))
49        .map(fs::read)?
50        .map_err(|e| anyhow!("Failed to read IDL `{name}`: {e}"))
51        .map(|buf| convert_idl(&buf))?
52}

54fn gen_program(idl: &Idl, name: &syn::Ident) -> proc_macro2::TokenStream {
55    let docs = gen_program_docs(idl);
56    let id = gen_id(idl);
57    let program_mod = gen_program_mod(&idl.metadata.name);
58
59    // Defined
60    let constants_mod = gen_constants_mod(idl);
61    let accounts_mod = gen_accounts_mod(idl);
62    let events_mod = gen_events_mod(idl);
63    let types_mod = gen_types_mod(idl);
64
65    // Clients
66    let cpi_mod = gen_cpi_mod(idl);
67    let client_mod = gen_client_mod(idl);
68    let internal_mod = gen_internal_mod(idl);
69
70    // Utils
71    let utils_mod = gen_utils_mod(idl);
72
73    quote! {
74        #docs
75        pub mod #name {
76            use anchor_lang::prelude::*;
77            use accounts::*;
78            use events::*;
79            use types::*;
80
81            #id
82            #program_mod
83
84            #constants_mod
85            #accounts_mod
86            #events_mod
87            #types_mod
88
89            #cpi_mod
90            #client_mod
91            #internal_mod
92
93            #utils_mod
94        }
95    }
96}

98fn gen_program_docs(idl: &Idl) -> proc_macro2::TokenStream {
99    let docs: &[String] = &[
100        format!(
101            "Generated external program declaration of program `{}`.",
102            idl.metadata.name
103        ),
104        String::default(),
105    ];
106    let docs = [docs, &idl.docs].concat();
107    gen_docs(&docs)
108}

110fn gen_id(idl: &Idl) -> proc_macro2::TokenStream {
111    let address_bytes = bs58::decode(&idl.address)
112        .into_vec()
113        .expect("Invalid `idl.address`");
114    let doc = format!("Program ID of program `{}`.", idl.metadata.name);
115
116    quote! {
117        #[doc = #doc]
118        pub static ID: Pubkey = __ID;
119
120        /// Const version of `ID`
121        pub const ID_CONST: Pubkey = __ID_CONST;
122
123        /// The name is intentionally prefixed with `__` in order to reduce to possibility of name
124        /// clashes with the crate's `ID`.
125        static __ID: Pubkey = Pubkey::new_from_array([#(#address_bytes,)*]);
126        const __ID_CONST : Pubkey = Pubkey::new_from_array([#(#address_bytes,)*]);
127    }
128}
