1use anchor_lang_idl::types::{Idl, IdlType};
2use quote::{format_ident, quote, ToTokens};
3
4use super::common::{convert_idl_type_to_syn_type, gen_docs};

6pub fn gen_constants_mod(idl: &Idl) -> proc_macro2::TokenStream {
7    let constants = idl.constants.iter().map(|c| {
8        let name = format_ident!("{}", c.name);
9        let docs = gen_docs(&c.docs);
10        let val = syn::parse_str::<syn::Expr>(&c.value)
11            .unwrap()
12            .to_token_stream();
13        let (ty, val) = match &c.ty {
14            IdlType::Bytes => (quote!(&[u8]), quote! { &#val }),
15            IdlType::String => (quote!(&str), val),
16            _ => (convert_idl_type_to_syn_type(&c.ty).to_token_stream(), val),
17        };
18
19        quote! {
20            #docs
21            pub const #name: #ty = #val;
22        }
23    });
24
25    quote! {
26        /// Program constants.
27        pub mod constants {
28            #(#constants)*
29        }
30    }
31}
