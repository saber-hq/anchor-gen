9// This struct describes the span of a `BitSlice` or `BitSliceMut`, starting
10// with of offset of `offset` bits into the array of blocks, and including
11// `len` bits.
12#[derive(Copy, Clone, Debug)]
13struct SliceSpan {
14    offset: u8,
15    len: u64,
16    aligned_blocks: usize,
17}
18// Invariant:
19//   aligned_blocks == if offset == 0 { Block::ceil_div_nbits(len) } else { 0 }
20
21// This struct describes the result of an indexing operation against a span.
22// We can give back a full, aligned block, or an arbitrary sequence of bits.
23#[derive(Copy, Clone, Debug)]
24enum BlockAddress {
25    FullBlockAt(usize),
26    SomeBitsAt(Address, usize),
27}
28
29impl SliceSpan {
30    fn new<Block: BlockType>(offset: u8, bit_len: u64) -> Self {
31        SliceSpan {
32            offset,
33            len: bit_len,
34            aligned_blocks: if offset == 0 {Block::ceil_div_nbits(bit_len)} else {0},
35        }
36    }
37
38    fn from_block_len<Block: BlockType>(block_len: usize) -> Self {
39        Self::new::<Block>(0, Block::mul_nbits(block_len))
40    }
41
42    fn block_len<Block: BlockType>(&self) -> usize {
43        Block::ceil_div_nbits(self.len)
44    }
45
46    fn find_block<Block: BlockType>(&self, position: usize) -> Option<BlockAddress> {
47        if position < self.aligned_blocks {
48            return Some(BlockAddress::FullBlockAt(position));
49        } else if position < self.block_len::<Block>() {
50            let start   = Block::mul_nbits(position) + u64::from(self.offset);
51            let address = Address::new::<Block>(start);
52            let count   = Block::block_bits(self.len, position);
53            Some(BlockAddress::SomeBitsAt(address, count))
54        } else {
55            None
56        }
57    }
58
59    fn find_bits<Block: BlockType>(&self, position: u64, count: usize) -> Option<BlockAddress> {
60        if position + (count as u64) <= self.len {
61            let address = Address::new::<Block>(position + u64::from(self.offset));
62            if count == Block::nbits() && address.bit_offset == 0 {
63                Some(BlockAddress::FullBlockAt(address.block_index))
64            } else {
65                Some(BlockAddress::SomeBitsAt(address, count))
66            }
67        } else {
68            None
69        }
70    }
71
72    fn find_bit<Block: BlockType>(&self, position: u64) -> Option<Address> {
73        if position < self.len {
74            Some(Address::new::<Block>(position + self.offset as u64))
75        } else {
76            None
77        }
78    }
79}

81impl BlockAddress {
82    unsafe fn read<Block: BlockType>(self, bits: *const Block) -> Block {
83        match self {
84            BlockAddress::FullBlockAt(position) =>
85                ptr::read(bits.offset(position as isize)),
86
87            BlockAddress::SomeBitsAt(address, count) => {
88                let offset      = address.bit_offset;
89                let ptr1        = bits.offset(address.block_index as isize);
90                let block1      = ptr::read(ptr1);
91
92                // Otherwise, our access is unaligned and may span two blocks. So we need
93                // to get our bits starting at `offset` in `block1`, and the rest from `block2`
94                // if necessary.
95                let shift1      = offset as usize;
96                let shift2      = Block::nbits() - shift1;
97
98                // Getting the right number of bits from `block1`.
99                let bits_size1  = cmp::min(shift2, count);
100                let chunk1      = block1.get_bits(shift1, bits_size1);
101
102                // The remaining bits will be in `block2`; if there are none, we return early.
103                let bits_size2  = count - bits_size1;
104                if bits_size2 == 0 {
105                    return chunk1;
106                }
107
108                // Otherwise, we need to get `block2` and combine bits from each block to get
109                // the result.
110                let block2      = ptr::read(ptr1.offset(1));
111                let chunk2      = block2.get_bits(0, bits_size2);
112                (chunk1 | (chunk2 << shift2))
113            }
114        }
115    }
116
117    unsafe fn write<Block: BlockType>(self, bits: *mut Block, value: Block) {
118        match self {
119            BlockAddress::FullBlockAt(position) =>
120                ptr::write(bits.offset(position as isize), value),
121
122            BlockAddress::SomeBitsAt(address, count) => {
123                let offset  = address.bit_offset;
124                let ptr1    = bits.offset(address.block_index as isize);
125
126                // Otherwise, our access is unaligned. In particular, we need to align
127                // the first bits of `value` with the last `Block::nbits() - offset` bits
128                // of `block1`, and the last `offset` bits of value with the first bits
129                // of `block2`.
130                let shift1      = offset;
131                let shift2      = Block::nbits() - shift1;
132
133                // We aren't necessarily going to keep all `shift2` bits of `value`, though,
134                // because it might exceed the `count`. In any case, we can now read the block,
135                // overwrite the correct bits, and write the block back.
136                let bits_size1  = cmp::min(count, shift2);
137                let old_block1  = ptr::read(ptr1);
138                let new_block1  = old_block1.with_bits(shift1, bits_size1, value);
139                ptr::write(ptr1, new_block1);
140
141                // The remaining bits to change in `block2`. If it's zero, we finish early.
142                let bits_size2  = count - bits_size1;
143                if bits_size2 == 0 { return; }
144                let ptr2        = ptr1.offset(1);
145                let old_block2  = ptr::read(ptr2);
146                let new_block2  = old_block2.with_bits(0, bits_size2, value >> shift2);
147                ptr::write(ptr2, new_block2);
148            }
149        }
150    }
151}

153/// A slice of a bit-vector; akin to `&'a [bool]` but packed.
154///
155/// # Examples
156///
157/// ```
158/// use bv::*;
159///
160/// let array = [0b00110101u16];
161/// let mut slice = array.bit_slice(..8);
162/// assert_eq!( slice[0], true );
163/// assert_eq!( slice[1], false );
164///
165/// slice = slice.bit_slice(1..8);
166/// assert_eq!( slice[0], false );
167/// assert_eq!( slice[1], true );
168/// ```
169#[derive(Copy, Clone)]
170pub struct BitSlice<'a, Block> {
171    bits:   *const Block,
172    span:   SliceSpan,
173    marker: PhantomData<&'a ()>,
174}
175
176/// A mutable slice of a bit-vector; akin to `&'a mut [bool]` but packed.
177///
178/// # Examples
179///
180/// ```
181/// use bv::*;
182///
183/// let mut array = [0b00110101u16];
184///
185/// {
186///     let mut slice = BitSliceMut::from_slice(&mut array);
187///     assert_eq!( slice[0], true );
188///     assert_eq!( slice[1], false );
189///
190///     slice.set_bit(0, false);
191/// }
192///
193/// assert_eq!( array[0], 0b00110100u16 );
194/// ```
195pub struct BitSliceMut<'a, Block> {
196    bits:   *mut Block,
197    span:   SliceSpan,
198    marker: PhantomData<&'a mut ()>,
199}

201impl<'a, Block: BlockType> BitSlice<'a, Block> {
202    /// Creates a `BitSlice` from an array slice of blocks.
203    ///
204    /// The size is always a multiple of
205    /// `Block::nbits()`. If you want a different size, slice.
206    ///
207    /// # Examples
208    ///
209    /// ```
210    /// use bv::{BitSlice, BitSliceable};
211    ///
212    /// let v = vec![0b01010011u16, 0u16];
213    /// let slice = BitSlice::from_slice(&v).bit_slice(..7);
214    /// assert_eq!( slice.len(), 7 );
215    /// assert_eq!( slice[0], true );
216    /// assert_eq!( slice[1], true );
217    /// assert_eq!( slice[2], false );
218    /// ```
219    pub fn from_slice(blocks: &'a [Block]) -> Self {
220        BitSlice {
221            bits:   blocks.as_ptr(),
222            span:   SliceSpan::from_block_len::<Block>(blocks.len()),
223            marker: PhantomData,
224        }
225    }
226
227    /// Creates a `BitSlice` from a pointer to its data, an offset where the bits start, and
228    /// the number of available bits.
229    ///
230    /// This is unsafe because the size of the passed-in buffer is
231    /// not checked. It must hold at least `offset + len` bits or the resulting behavior is
232    /// undefined.
233    ///
234    /// # Precondition
235    ///
236    ///   - the first `Block::ceil_div_nbits(len + offset)` words of `bits` safe
237    ///     to read.
238    pub unsafe fn from_raw_parts(bits: *const Block, offset: u64, len: u64) -> Self {
239        let address = Address::new::<Block>(offset);
240        BitSlice {
241            bits:   bits.offset(address.block_index as isize),
242            span:   SliceSpan::new::<Block>(address.bit_offset as u8, len),
243            marker: PhantomData,
244        }
245    }
246
247    /// The number of bits in the slice.
248    ///
249    /// # Examples
250    ///
251    /// ```
252    /// use bv::*;
253    ///
254    /// let bv: BitVec = bit_vec![ true, true, false, true ];
255    /// let slice = bv.bit_slice(..3);
256    ///
257    /// assert_eq!( bv.len(), 4 );
258    /// assert_eq!( slice.len(), 3 );
259    /// ```
260    pub fn len(&self) -> u64 {
261        self.span.len
262    }
263
264    /// Returns whether there are no bits in the slice.
265    ///
266    /// # Examples
267    ///
268    /// ```
269    /// use bv::*;
270    ///
271    /// let bv: BitVec = bit_vec![ true, true, false, true ];
272    /// let slice0 = bv.bit_slice(3..3);
273    /// let slice1 = bv.bit_slice(3..4);
274    ///
275    /// assert!(  slice0.is_empty() );
276    /// assert!( !slice1.is_empty() );
277    /// ```
278    pub fn is_empty(&self) -> bool {
279        self.len() == 0
280    }
281}

354// Gets `bits[offset + position]`.
355//
356// Precondition: Bits are in bounds.
357unsafe fn get_raw_bit<Block: BlockType>(bits: *const Block, address: Address) -> bool {
358
359    let ptr       = bits.offset(address.block_index as isize);
360    let block     = ptr::read(ptr);
361    block.get_bit(address.bit_offset)
362}
363
364// Sets `bits[offset + position]`
365//
366// Precondition: Bit is in bounds.
367unsafe fn set_raw_bit<Block: BlockType>(bits: *mut Block, address: Address, value: bool) {
368    let ptr       = bits.offset(address.block_index as isize);
369    let old_block = ptr::read(ptr);
370    let new_block = old_block.with_bit(address.bit_offset, value);
371    ptr::write(ptr, new_block);
372}

653impl<'a, Block: BlockType + hash::Hash> hash::Hash for BitSlice<'a, Block> {
654    fn hash<H: hash::Hasher>(&self, state: &mut H) {
655        state.write_u64(self.bit_len());
656        for block in BlockIter::new(self) {
657            block.hash(state);
658        }
659    }
660}
661
662impl<'a, Block: BlockType + hash::Hash> hash::Hash for BitSliceMut<'a, Block> {
663    fn hash<H: hash::Hasher>(&self, state: &mut H) {
664        self.as_bit_slice().hash(state);
665    }
666}

668impl<'a, Block: BlockType> fmt::Debug for BitSlice<'a, Block> {
669    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
670        write!(f, "bit_vec![")?;
671        if !self.is_empty() {
672            write!(f, "{}", self.get_bit(0))?;
673        }
674        for i in 1 .. self.span.len {
675            write!(f, ", {}", self.get_bit(i))?;
676        }
677        write!(f, "]")
678    }
679}

681impl<'a, Block: BlockType> fmt::Debug for BitSliceMut<'a, Block> {
682    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
683        self.as_bit_slice().fmt(f)
684    }
