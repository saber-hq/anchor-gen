6// Generates the `Accounts` trait implementation.
7pub fn generate(accs: &AccountsStruct) -> proc_macro2::TokenStream {
8    let name = &accs.ident;
9    let ParsedGenerics {
10        combined_generics,
11        trait_generics,
12        struct_generics,
13        where_clause,
14    } = generics(accs);
15
16    // Deserialization for each field
17    let deser_fields: Vec<proc_macro2::TokenStream> = accs
18        .fields
19        .iter()
20        .map(|af: &AccountField| {
21            match af {
22                AccountField::CompositeField(s) => {
23                    let name = &s.ident;
24                    let ty = &s.raw_field.ty;
25                    quote! {
26                        #[cfg(feature = "anchor-debug")]
27                        ::solana_program::log::sol_log(stringify!(#name));
28                        let #name: #ty = anchor_lang::Accounts::try_accounts(__program_id, __accounts, __ix_data, &mut __bumps.#name, __reallocs)?;
29                    }
30                }
31                AccountField::Field(f) => {
32                    // `init` and `zero` accounts are special cased as they are
33                    // deserialized by constraints. Here, we just take out the
34                    // AccountInfo for later use at constraint validation time.
35                    if is_init(af) || f.constraints.zeroed.is_some()  {
36                        let name = &f.ident;
37                        // Optional accounts have slightly different behavior here and
38                        // we can't leverage the try_accounts implementation for zero and init.
39                        if f.is_optional {
40                            // Thus, this block essentially reimplements the try_accounts 
41                            // behavior with optional accounts minus the deserialziation.
42                            let empty_behavior = if cfg!(feature = "allow-missing-optionals") {
43                                quote!{ None }
44                            } else {
45                                quote!{ return Err(anchor_lang::error::ErrorCode::AccountNotEnoughKeys.into()); }
46                            };
47                            quote! {
48                                let #name = if __accounts.is_empty() {
49                                    #empty_behavior
50                                } else if __accounts[0].key == __program_id {
51                                    *__accounts = &__accounts[1..];
52                                    None
53                                } else {
54                                    let account = &__accounts[0];
55                                    *__accounts = &__accounts[1..];
56                                    Some(account)
57                                };
58                            }
59                        } else {
60                            quote!{
61                                if __accounts.is_empty() {
62                                    return Err(anchor_lang::error::ErrorCode::AccountNotEnoughKeys.into());
63                                }
64                                let #name = &__accounts[0];
65                                *__accounts = &__accounts[1..];
66                            }
67                        }
68                    } else {
69                        let name = f.ident.to_string();
70                        let typed_name = f.typed_ident();
71                        quote! {
72                            #[cfg(feature = "anchor-debug")]
73                            ::solana_program::log::sol_log(stringify!(#typed_name));
74                            let #typed_name = anchor_lang::Accounts::try_accounts(__program_id, __accounts, __ix_data, __bumps, __reallocs)
75                                .map_err(|e| e.with_account_name(#name))?;
76                        }
77                    }
78                }
79            }
80        })
81        .collect();
82
83    let constraints = generate_constraints(accs);
84    let accounts_instance = generate_accounts_instance(accs);
85    let bumps_struct_name = bumps::generate_bumps_name(&accs.ident);
86
87    let ix_de = match &accs.instruction_api {
88        None => quote! {},
89        Some(ix_api) => {
90            let strct_inner = &ix_api;
91            let field_names: Vec<proc_macro2::TokenStream> = ix_api
92                .iter()
93                .map(|expr: &Expr| match expr {
94                    Expr::Type(expr_type) => {
95                        let field = &expr_type.expr;
96                        quote! {
97                            #field
98                        }
99                    }
100                    _ => panic!("Invalid instruction declaration"),
101                })
102                .collect();
103            quote! {
104                let mut __ix_data = __ix_data;
105                #[derive(anchor_lang::AnchorSerialize, anchor_lang::AnchorDeserialize)]
106                struct __Args {
107                    #strct_inner
108                }
109                let __Args {
110                    #(#field_names),*
111                } = __Args::deserialize(&mut __ix_data)
112                    .map_err(|_| anchor_lang::error::ErrorCode::InstructionDidNotDeserialize)?;
113            }
114        }
115    };
116
117    quote! {
118        #[automatically_derived]
119        impl<#combined_generics> anchor_lang::Accounts<#trait_generics, #bumps_struct_name> for #name<#struct_generics> #where_clause {
120            #[inline(never)]
121            fn try_accounts(
122                __program_id: &anchor_lang::solana_program::pubkey::Pubkey,
123                __accounts: &mut &#trait_generics [anchor_lang::solana_program::account_info::AccountInfo<#trait_generics>],
124                __ix_data: &[u8],
125                __bumps: &mut #bumps_struct_name,
126                __reallocs: &mut std::collections::BTreeSet<anchor_lang::solana_program::pubkey::Pubkey>,
127            ) -> anchor_lang::Result<Self> {
128                // Deserialize instruction, if declared.
129                #ix_de
130                // Deserialize each account.
131                #(#deser_fields)*
132                // Execute accounts constraints.
133                #constraints
134                // Success. Return the validated accounts.
135                Ok(#accounts_instance)
136            }
137        }
138    }
139}

141pub fn generate_constraints(accs: &AccountsStruct) -> proc_macro2::TokenStream {
142    let non_init_fields: Vec<&AccountField> =
143        accs.fields.iter().filter(|af| !is_init(af)).collect();
144
145    // Deserialization for each pda init field. This must be after
146    // the initial extraction from the accounts slice and before access_checks.
147    let init_fields: Vec<proc_macro2::TokenStream> = accs
148        .fields
149        .iter()
150        .filter_map(|af| match af {
151            AccountField::CompositeField(_s) => None,
152            AccountField::Field(f) => match is_init(af) {
153                false => None,
154                true => Some(f),
155            },
156        })
157        .map(|f| constraints::generate(f, accs))
158        .collect();
159
160    // Constraint checks for each account fields.
161    let access_checks: Vec<proc_macro2::TokenStream> = non_init_fields
162        .iter()
163        .map(|af: &&AccountField| match af {
164            AccountField::Field(f) => constraints::generate(f, accs),
165            AccountField::CompositeField(s) => constraints::generate_composite(s),
166        })
167        .collect();
168
169    quote! {
170        #(#init_fields)*
171        #(#access_checks)*
172    }
173}

175pub fn generate_accounts_instance(accs: &AccountsStruct) -> proc_macro2::TokenStream {
176    let name = &accs.ident;
177    // Each field in the final deserialized accounts struct.
178    let return_tys: Vec<proc_macro2::TokenStream> = accs
179        .fields
180        .iter()
181        .map(|f: &AccountField| {
182            let name = match f {
183                AccountField::CompositeField(s) => &s.ident,
184                AccountField::Field(f) => &f.ident,
185            };
186            quote! {
187                #name
188            }
189        })
190        .collect();
191
192    quote! {
193        #name {
194            #(#return_tys),*
195        }
196    }
197}
198
199fn is_init(af: &AccountField) -> bool {
200    match af {
201        AccountField::CompositeField(_s) => false,
202        AccountField::Field(f) => f.constraints.init.is_some(),
203    }
