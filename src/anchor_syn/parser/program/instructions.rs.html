9// Parse all non-state ix handlers from the program mod definition.
10pub fn parse(program_mod: &syn::ItemMod) -> ParseResult<(Vec<Ix>, Option<FallbackFn>)> {
11    let mod_content = &program_mod
12        .content
13        .as_ref()
14        .ok_or_else(|| ParseError::new(program_mod.span(), "program content not provided"))?
15        .1;
16
17    let ixs = mod_content
18        .iter()
19        .filter_map(|item| match item {
20            syn::Item::Fn(item_fn) => {
21                let (ctx, _) = parse_args(item_fn).ok()?;
22                ctx_accounts_ident(&ctx.raw_arg).ok()?;
23                Some(item_fn)
24            }
25            _ => None,
26        })
27        .map(|method: &syn::ItemFn| {
28            let (ctx, args) = parse_args(method)?;
29            let overrides = parse_overrides(&method.attrs)?;
30            let interface_discriminator = spl_interface::parse(&method.attrs);
31            let docs = docs::parse(&method.attrs);
32            let cfgs = parse_cfg(method);
33            let returns = parse_return(method)?;
34            let anchor_ident = ctx_accounts_ident(&ctx.raw_arg)?;
35            Ok(Ix {
36                raw_method: method.clone(),
37                ident: method.sig.ident.clone(),
38                docs,
39                cfgs,
40                args,
41                anchor_ident,
42                returns,
43                interface_discriminator,
44                overrides,
45            })
46        })
47        .collect::<ParseResult<Vec<Ix>>>()?;
48
49    let fallback_fn = {
50        let fallback_fns = mod_content
51            .iter()
52            .filter_map(|item| match item {
53                syn::Item::Fn(item_fn) => {
54                    let (ctx, _args) = parse_args(item_fn).ok()?;
55                    if ctx_accounts_ident(&ctx.raw_arg).is_ok() {
56                        return None;
57                    }
58                    Some(item_fn)
59                }
60                _ => None,
61            })
62            .collect::<Vec<_>>();
63        if fallback_fns.len() > 1 {
64            return Err(ParseError::new(
65                fallback_fns[0].span(),
66                "More than one fallback function found",
67            ));
68        }
69        fallback_fns
70            .first()
71            .map(|method: &&syn::ItemFn| FallbackFn {
72                raw_method: (*method).clone(),
73            })
74    };
75
76    Ok((ixs, fallback_fn))
77}

79/// Parse overrides from the `#[instruction]` attribute proc-macro.
80fn parse_overrides(attrs: &[syn::Attribute]) -> ParseResult<Option<Overrides>> {
81    attrs
82        .iter()
83        .find(|attr| match attr.path.segments.last() {
84            Some(seg) => seg.ident == "instruction",
85            _ => false,
86        })
87        .map(|attr| attr.parse_args())
88        .transpose()
89}

91pub fn parse_args(method: &syn::ItemFn) -> ParseResult<(IxArg, Vec<IxArg>)> {
92    let mut args: Vec<IxArg> = method
93        .sig
94        .inputs
95        .iter()
96        .map(|arg: &syn::FnArg| match arg {
97            syn::FnArg::Typed(arg) => {
98                let docs = docs::parse(&arg.attrs);
99                let ident = match &*arg.pat {
100                    syn::Pat::Ident(ident) => &ident.ident,
101                    _ => return Err(ParseError::new(arg.pat.span(), "expected argument name")),
102                };
103                Ok(IxArg {
104                    name: ident.clone(),
105                    docs,
106                    raw_arg: arg.clone(),
107                })
108            }
109            syn::FnArg::Receiver(_) => Err(ParseError::new(
110                arg.span(),
111                "expected a typed argument not self",
112            )),
113        })
114        .collect::<ParseResult<_>>()?;
115
116    // Remove the Context argument
117    let ctx = args.remove(0);
118
119    Ok((ctx, args))
120}

122pub fn parse_return(method: &syn::ItemFn) -> ParseResult<IxReturn> {
123    match method.sig.output {
124        syn::ReturnType::Type(_, ref ty) => {
125            let ty = match ty.as_ref() {
126                syn::Type::Path(ty) => ty,
127                _ => return Err(ParseError::new(ty.span(), "expected a return type")),
128            };
129            // Assume unit return by default
130            let default_generic_arg = syn::GenericArgument::Type(syn::parse_str("()").unwrap());
131            let generic_args = match &ty.path.segments.last().unwrap().arguments {
132                syn::PathArguments::AngleBracketed(params) => {
133                    params.args.iter().next_back().unwrap()
134                }
135                _ => &default_generic_arg,
136            };
137            let ty = match generic_args {
138                syn::GenericArgument::Type(ty) => ty.clone(),
139                _ => {
140                    return Err(ParseError::new(
141                        ty.span(),
142                        "expected generic return type to be a type",
143                    ))
144                }
145            };
146            Ok(IxReturn { ty })
147        }
148        _ => Err(ParseError::new(
149            method.sig.output.span(),
150            "expected a return type",
151        )),
152    }
153}

155fn parse_cfg(method: &syn::ItemFn) -> Vec<Attribute> {
156    method
157        .attrs
158        .iter()
159        .filter_map(|attr| match attr.path.is_ident("cfg") {
160            true => Some(attr.to_owned()),
161            false => None,
162        })
163        .collect()
164}
